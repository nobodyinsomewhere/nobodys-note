    function download(filename, content, type = "text/plain;charset=utf-8") {
      const blob = new Blob([content], { type });
      const url = URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = filename;
      a.click();
      setTimeout(() => URL.revokeObjectURL(url), 1500);
    }

    async function copyText(text) {
      await navigator.clipboard.writeText(text);
    }

    function generate() {
      const title = $("postTitle").value.trim();
      const date = $("postDate").value.trim();
      const fileName = $("fileName").value.trim();
      const excerpt = $("postExcerpt").value.trim();
      const body = $("postBody").value.trim();
      const theme = $("themeSelect").value;
      const postsRaw = $("postsJsonInput").value.trim();

      if (!title || !date || !fileName || !excerpt || !body) {
        alert("请先把标题、日期、文件名、摘要和正文都填完整。");
        return;
      }

      if (!postsRaw) {
        alert("请先把当前的 posts.json 内容粘进来。");
        return;
      }

      const currentPosts = safeJsonParse(postsRaw, null);
      if (!Array.isArray(currentPosts)) {
        alert("posts.json 内容格式不正确，必须是 JSON 数组。");
        return;
      }

      const bodyParagraphs = splitParagraphs(body);

      const articleHtml = buildArticleHtml({
        title,
        date,
        theme,
        bodyParagraphs
      });

      const newItem = {
        title,
        date,
        excerpt,
        file: fileName
      };

      const nextPosts = [
        newItem,
        ...currentPosts.filter(item => item.file !== fileName)
      ];

      $("articleOutput").textContent = articleHtml;
      $("postsOutput").textContent = JSON.stringify(nextPosts, null, 2);
    }

    $("generateBtn").addEventListener("click", generate);

        return;

      alert("文章 HTML 已复制。");
    });

    $("downloadArticleBtn").addEventListener("click", () => {

        return;
      }

    });
